  </style>

<body>

  <h1>FITIFY</h1>

  <div class="section">

    <label>Choose an option:</label><br>

    <button onclick="showMoodForm()">1. Suggest workout by mood</button>

    <button onclick="showPlanForm()">2. Generate weekly workout plan</button>

  </div>

  <div class="section" id="moodForm" style="display:none;">

    <label>Enter your mood:</label>

    <input type="text" id="moodInput" placeholder="e.g., Happy, Stressed" />

    <button onclick="suggestWorkout()">Suggest</button>

  </div>

  <div class="section" id="planForm" style="display:none;">

    <label>What's your goal?</label><br>

    <select id="goalSelect">

      <option value="1">Lose weight</option>

      <option value="2">Gain weight</option>

    </select><br>

    <label>How many days per week?</label>

    <input type="number" id="daysInput" min="1" max="7" />

    <button onclick="generatePlan()">Generate</button>

  </div>

  <div class="output" id="result"></div>

  <script>

    function showMoodForm() {

      document.getElementById('moodForm').style.display = 'block';

      document.getElementById('planForm').style.display = 'none';

      document.getElementById('result').innerHTML = '';

    }

    function showPlanForm() {

      document.getElementById('planForm').style.display = 'block';

      document.getElementById('moodForm').style.display = 'none';

      document.getElementById('result').innerHTML = '';

    }

    function suggestWorkout() {

      const mood = document.getElementById('moodInput').value.toLowerCase();

      let output = '';

      switch (mood) {

        case 'happy': output = "Suggested workout: Yoga or Dance."; break;

        case 'stressed': output = "Suggested workout: Meditation or Light Stretching."; break;

        case 'energetic': output = "Suggested workout: HIIT or Running."; break;

        case 'lazy': output = "Suggested workout: Walking or Light Stretching."; break;

        default: output = "Mood not recognized. Try Yoga for balance.";

      }

      document.getElementById('result').innerHTML = output;

    }

    function generatePlan() {

      const goal = parseInt(document.getElementById('goalSelect').value);

      const days = parseInt(document.getElementById('daysInput').value);

      if (days < 1 || days > 7) {

        document.getElementById('result').innerHTML = "Please enter a number between 1 and 7.";

        return;

      }

      let output = "";

      if (days === 1) {

        output = `

          <b>Full Body Workout:</b><br>

          Chest: Bench Press (3x8-12)<br>

          Back: Pull-ups, Rows (3x8-12)<br>

          Legs: Squats (3x8-12)<br>

          Shoulders: Press (3x8-12)<br>

          Arms: Curl, Kickbacks (3x8-12)<br>`;

        if (goal === 1) output += "Cardio + Plank";

      } else if (days === 2) {

        output = `

          <b>Day 1:</b> Chest, Triceps, Shoulders, Legs<br>

          - Machine Chest Press, Incline DB Press, Cable Fly<br>

          - Triceps: Pushdown, Overhead Ext<br>

          - Shoulders: Press, Raises<br>

          - Legs: Squats<br><br>

          <b>Day 2:</b> Back, Biceps, Forearms, Legs<br>

          - Pull-ups, Lat Pulldown, Rows<br>

          - Biceps: Curl, Hammer<br>

          - Forearms: Wrist Curls<br>

          - Legs: Leg Ext<br>`;

        if (goal === 1) output += "Cardio + Abs";

      } else {

        const splits = [

          ["Chest and Triceps", "Chest Press, DB Press, Cable Fly, Kickbacks, Pushdowns"],

          ["Back, Biceps, Forearms", "Pull-ups, Rows, Curls, Hammer, Wrist Curls"],

          ["Legs and Shoulders", "Squats, Leg Ext, Press, Raises"]

        ];

        output += "<b>Weekly Split:</b><br>";

        for (let i = 0; i < days; i++) {

          const group = splits[i % splits.length];

          output += `Day ${i+1} (${group[0]}): ${group[1]}<br>`;

          if (goal === 1 && i % splits.length === 2) output += "Cardio + Abs<br>";

        }

      }

      document.getElementById('result').innerHTML = output;

    }

  </script>
